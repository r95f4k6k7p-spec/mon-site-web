  <!-- Scripts simples: mobile nav, smooth scroll, validation -->

  <script>

    // Year

    document.getElementById('year').textContent = new Date().getFullYear();


    // Mobile nav toggle

    const hambtn = document.getElementById('hambtn');

    const mobnav = document.getElementById('mobnav');

    hambtn.addEventListener('click', () => {

      const open = mobnav.style.display === 'block';

      mobnav.style.display = open ? 'none' : 'block';

      hambtn.setAttribute('aria-expanded', String(!open));

    });


    // Smooth scroll for internal links

    document.querySelectorAll('a[href^="#"]').forEach(a=>{

      a.addEventListener('click', (e)=>{

        const target = document.querySelector(a.getAttribute('href'));

        if(target){

          e.preventDefault();

          target.scrollIntoView({behavior:'smooth', block:'start'});

          // close mobile nav on selection

          if(window.innerWidth <= 640) mobnav.style.display = 'none';

        }

      });

    });


    // Simple contact form handling (front-end only)

    const form = document.getElementById('contactForm');

    const formMsg = document.getElementById('formMsg');

    form.addEventListener('submit', (e) => {

      e.preventDefault();

      formMsg.textContent = '';

      const name = document.getElementById('name').value.trim();

      const email = document.getElementById('email').value.trim();

      const message = document.getElementById('message').value.trim();


      // validation: very simple, accessible feedback

      if(!name || !email || !message){

        formMsg.style.color = '#fca5a5';

        formMsg.textContent = 'Merci de remplir tous les champs.';

        return;

      }

      if(!/^[^\s@]+@[^\s@]+\.[^\s@]+$/.test(email)){

        formMsg.style.color = '#fca5a5';

        formMsg.textContent = 'Merci d\'indiquer un email valide.';

        return;

      }


      // simulate success (replace with real request to your backend)

      formMsg.style.color = '';

      formMsg.textContent = 'Message envoyé — merci ! (Simulation front-end)';

      form.reset();

    });

  </script>
